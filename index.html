from binance.client import Client
from binance.enums import *
import pandas as pd
import ta, time

# 👇 Buraya kendi anahtarlarını yaz
API_KEY = "BURAYA_API_KEY"
SECRET_KEY = "BURAYA_SECRET_KEY"
SYMBOL = "BTCUSDT"
TRADE_USDT = 20 # Her işlemde 20 USDT kullan
STOP_LOSS = 0.02 # %2 zarar durdur
TAKE_PROFIT = 0.04 # %4 kar al

client = Client(API_KEY, SECRET_KEY, testnet=True)

def get_signal():
    klines = client.get_klines(symbol=SYMBOL, interval=Client.KLINE_INTERVAL_15MINUTE, limit=100)
    df = pd.DataFrame(klines, columns=['t','o','h','l','c','v','ct','qa','n','tb','tq','i'])
    df['close'] = pd.to_numeric(df['c'])
    df['rsi'] = ta.momentum.RSIIndicator(df['close'], window=14).rsi()
    df['ma20'] = df['close'].rolling(20).mean()
    last = df.iloc[-1]
    if last['rsi'] < 35 and last['close'] > last['ma20']: return "BUY"
    if last['rsi'] > 65 and last['close'] < last['ma20']: return "SELL"
    return "HOLD"

def run_bot():
    print("✅ Bot başladı! Sinyal bekleniyor...")
    while True:
        signal = get_signal()
        print(f"Sinyal: {signal}")
        # Gerçek emir gönderme kısmı burada olur
        time.sleep(60) # Her dakika kontrol et

run_bot()
